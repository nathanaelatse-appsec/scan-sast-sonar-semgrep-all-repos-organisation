#!/bin/bash

GITHUB_ORG="nom_organisation"
GITHUB_TOKEN="votre_token_github"
WORK_DIR="/repos/downloaded/directory"
PAGE=1
CLONED=0
FAILED=0

mkdir -p $WORK_DIR

echo "=== Début du clonage $(date) ==="

while true; do
    echo "📄 Récupération page $PAGE..."

    # Récupérer les repos page par page (100 par page max)
    repos=$(curl -s -H "Authorization: token $GITHUB_TOKEN" \
        "https://api.github.com/orgs/$GITHUB_ORG/repos?per_page=100&page=$PAGE" \
        | python3 -c "
import json, sys
data = json.load(sys.stdin)
if not data:
    exit(1)
for r in data:
    print(r['clone_url'])
" 2>/dev/null)

    # Si plus de repos, on arrête
    if [ $? -ne 0 ] || [ -z "$repos" ]; then
        echo "✅ Plus de repos à récupérer."
        break
    fi

    # Cloner chaque repo
    for repo_url in $repos; do
        PROJECT_NAME=$(basename "$repo_url" .git)
        
        if [ -d "$WORK_DIR/$PROJECT_NAME" ]; then
            echo "⏭️  Déjà cloné, mise à jour : $PROJECT_NAME"
            git -C "$WORK_DIR/$PROJECT_NAME" pull --quiet
        else
            echo "📦 Clonage : $PROJECT_NAME"
            git clone --quiet "$repo_url" "$WORK_DIR/$PROJECT_NAME"
            if [ $? -eq 0 ]; then
                ((CLONED++))
            else
                echo "❌ Echec : $PROJECT_NAME"
                ((FAILED++))
            fi
        fi
    done

    ((PAGE++))
done

echo ""
echo "=== Résumé ==="
echo "✅ Clonés    : $CLONED"
echo "❌ Echoués   : $FAILED"
echo "📁 Dossier   : $WORK_DIR"
echo "=== Fin $(date) ==="
EOF
